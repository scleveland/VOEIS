def create
  parent.managed_repository do
    @data_set = Voeis::DataSet.new(params[:data_set])
    respond_to do |format|
      if @data_set.save
        flash[:notice] = 'DataSet  was successfully created.'
         format.json do
            render :json => @data_set.as_json, :callback => params[:jsoncallback]
          end
        format.html { (redirect_to(new_project_data_set_path())) }
        format.js
      else
        format.html { render :action => "new" }
      end
    end
  end
end

def opts_for_select(opt_array, selected = nil)
   option_string =""
   if !opt_array.empty?
     opt_array.each do |opt|
       if opt[1] == selected
         option_string = option_string + '<option selected="selected" value='+opt[1]+'>'+opt[0]+'</option>'
       else
         option_string = option_string + '<option value='+opt[1]+'>'+opt[0]+'</option>'
       end
     end
   end
   option_string
end
